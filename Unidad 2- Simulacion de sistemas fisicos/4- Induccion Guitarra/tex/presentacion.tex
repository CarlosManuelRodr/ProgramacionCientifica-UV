\begin{frame}{Sobre los modelos teóricos}
  \begin{itemize}
    \item Se pueden utilizar los campos de solenoides rectos finitos para modelar los imanes de las pastillas (N. G. Horton, T. R. Moore)
      \begin{itemize}
      \item Es matemáticamente complicado
      \item No aporta más información física acerca del problema que los demás métodos.
      \end{itemize}
    \item Se pueden modelar estas barras como simples dipolos (licenciatura)
      \begin{itemize}
      \item Mucho más sencilla de utilizar
      \item Válida en las regiones de interés (fuera del imán)
      \end{itemize}
    \item Campo magnético de cuerda vibrante en campo externo constante.
    \item Modelos realistas requieren soluciones numéricas (simulación)
  \end{itemize}

\begin{frame}{Un modelo más realista}
  \begin{columns}
    \begin{column}{0.5\textwidth}
      \begin{itemize}
        \item Se supondrá que la cuerda está inmersa en un campo magnético constante $B_0$ que permea todo el espacio.
        \item Al moverse la cuerda modifica el campo magnético.
        \item El flujo de este campo magnético modificado se induce en el solenoide.
      \end{itemize}
    \end{column}
    \begin{column}{0.5\textwidth}
      \begin{figure}
        \includegraphics[scale=0.25]{img/Diagrama.png}
        \caption{Diagrama del sistema}
      \end{figure}
    \end{column}
  \end{columns}
\end{frame}

%-----------------------------------------------------------------------------
\begin{frame}
  \begin{itemize}
  \item Se parte de la suposición de que no hay corriente en la cuerda (material conductor), entonces $\nabla \times \vec{H} = 0$.
  \item El campo magnético se puede expresar en términos de un potencial escalar $\vec{H} = - \nabla \phi_m$.
  \item El campo magnético que proviene del imán es
  \[
  	\vec{H_0} = \frac{B_0}{\mu_0} \hat{j} = - \nabla \frac{- B_0 r \sin \theta}{\mu_0}.
  \]
  \end{itemize}
\end{frame}

%-----------------------------------------------------------------------------
\begin{frame}
  Nos interesa el campo magnético de se genera de la interacción con la cuerda, se resuelve la ecuación de Laplace en coordenadas cilíndricas (por la geometría de la cuerda). El potencial no depende de $z$, se usa una forma conveniente de la solución a la ecuación de Laplace.
  \[
  	\phi_m(r, \theta) = A + B \operatorname{ln}(r) + \sum_{n=0}^\infty \left( A_n r^n \sin (n \theta) + B_n r^n \cos (n \theta) + C_n r^{-n} \sin (n \theta) \right.
  \]
  \[
  		\left. + D_n r^{-n} \cos (n \theta) \right)
  \]
\end{frame}

%-----------------------------------------------------------------------------
\begin{frame}{Caso $r > a$}
  \begin{itemize}
  \item No hay dependencia de los términos $\cos$.
  \item Al hacer $r \to \infty$ el campo que se observa es $H_0$.
  \item En la frontera $r=a$ el potencial es cero.
  \item A partir de estas condiciones el potencial se puede expresar como
  \[
  	\phi_m (r, \theta) = - \frac{B_0 r}{\mu_0} \sin \theta + \alpha \frac{a}{r} \sin \theta.
  \]
  \end{itemize}
\end{frame}

%-----------------------------------------------------------------------------
\begin{frame}{Caso $r < a$}
  \begin{itemize}
  \item No puede diverger en $r=0$.
  \item No tiene dependencia de los términos $\cos$.
  \item Sujeto a condiciones de frontera.
  \item A partir de estas condiciones el potencial se puede expresar como
  \[
  	\phi_m (r, \theta) = - \frac{B_0 r}{\mu_0} \sin \theta + \alpha \frac{r}{a} \sin \theta.
  \]
  \end{itemize}
\end{frame}

%-----------------------------------------------------------------------------
\begin{frame}
  El campo magnético normal a la superficie debe cumplir la condición de continuidad
  \[
  	B_r(r = a^{-}) = B_r(r = a^+),
  \]
  donde
  \[
  	B_r(r = a^-) = \mu_r \mu_0 H_r(r = a^-) = - \mu_r \mu_0 \left. \frac{\partial \phi_m}{\partial r} \right|_{r=a^-} = \left(B_0 - \frac{\mu_0 \alpha}{a} \right) \mu_r \sin \theta.
  \]
  y
  \[
  	B_r(r = a^+) = \mu_0 H_r(r = a^+) = - \mu_0 \left. \frac{\partial \phi_m}{\partial r} \right|_{r=a^+} = \left(B_0 + \frac{\mu_0 \alpha}{a} \right) \sin \theta.
  \]
  $\mu_r$ es la permeabilidad relativa de la cuerda.
\end{frame}

%-----------------------------------------------------------------------------
\begin{frame}
  Con esta condición de continuidad se obtiene el valor de $\alpha$,
  \[
  	\alpha = \frac{a B_0}{\mu_0} \frac{\mu_r - 1}{\mu_r + 1}.
  \]
  Al sustituir en la expresión para el potencial en el caso $r > a$ queda
  \[
  	\phi_m(r, \theta) = - \frac{B_0}{\mu_0} \left(r - \frac{a^2}{r} \frac{\mu_r - 1}{\mu_r + 1} \right) \sin \theta.
  \]
  A partir de aquí se obtiene el campo eléctrico radial y angular.
  \[
  	B_r(r, \theta) = - \mu_0 \frac{\partial \phi}{\partial r} = B_0 \left(1+\frac{a^2}{r^2} \frac{\mu_r - 1}{\mu_r + 1} \right) \sin \theta,
  \]
\end{frame}

%-----------------------------------------------------------------------------
\begin{frame}
  y
  \[
  	B_\theta (r, \theta) = - \frac{\mu_0}{r} \frac{\partial \phi}{\partial \theta} = B_0 \left(1 - \frac{a^2}{r^2} \frac{\mu_r - 1}{\mu_r+1} \right) \cos \theta.
  \]
  Como se desea calcular el campo magnético en la bobina se necesita el campo en dirección $y$.
  \[
  	B_y (x, y) = B_r \sin \theta + B_\theta \cos \theta = B_0 \left( 1 + a^2 \frac{\mu_r - 1}{\mu_r+1} \frac{y^2 - x^2}{(x^2 + y^2)^2} \right)
  \]
\end{frame}

\begin{frame}{Regresando al diagrama}
  \begin{columns}
    \begin{column}{0.5\textwidth}
      \begin{itemize}
        \item Se ubica a la bobina en la posición $(x_b, y_b)$.
        \item Se utilizará una bobina cuadrada que se extiende des de $x_b - \frac{L}{2}$ a $x_b - \frac{L}{2}$.
        \item La bobina tiene una separación $h$ de la cuerda en reposo.
      \end{itemize}
    \end{column}

%-----------------------------------------------------------------------------
\begin{frame}
  el flujo magnético en la bobina es
  \[
  	\Phi = \int \int \vec{B} \cdot d \vec{s} = L \int_{x_b-\frac{L}{2}}^{x_b + \frac{L}{2}} B_y(x, y_b) \, dx
  \]
  \[
  	= B_0 L \left( L + a^2 \frac{\mu_r - 1}{\mu_r+1} \left( \frac{x_b + \frac{L}{2}}{ \left(x_b + \frac{L}{2} \right)^2 + y_b} - \frac{x_b - \frac{L}{2}}{ \left(x_b - \frac{L}{2} \right)^2 + y_b} \right) \right).
  \]
\end{frame}

%-----------------------------------------------------------------------------
\begin{frame}
  Por conveniencia se pasa al sistema de referencia de la bobina, donde $x_b = x$, y $y_b = y + h$.
  \[
  	\Phi= B_0 L \left( L + a^2 \frac{\mu_r - 1}{\mu_r+1} \frac{(y+h)^2 - x^2 + \frac{L^2}{4}}{\left(x^2 - \frac{L^4}{4} \right)^2 + 2\left(x^2 + \frac{L^2}{4} \right)(y+h)^2 + (y+h)^4} \right),
  \]
  entonces el voltaje que se induce en la bobina es
  \small
  \[\Scale[0.8]{
  	V(t) = - \frac{d \Phi}{d t} = B_0 a^2 L^2 \frac{\mu_r - 1}{\mu_r+1} \left( \frac{2 x \dot{x} (3(y+h)^4 + 2(x^2 + L^2/4)(y+h)^2 - (x^2 - L^2/4)^2) + \dot{y}(2(y+h)^5 - 4(x^2 - L^2/4)(y+h)^3 + (6x^4 - x^2 L^2/2 - L^4/8)}{((x^2-L^2/4)^2 + 2(x^2 + L^2/4)(y+h)^2 + (y+h)^4} \right)}
  \]
  \[
  	\approx B_0 a^2 L^2 \frac{\mu_r - 1}{\mu_r+1} \left( 2 x \dot{x} \frac{3h^2 - L^2/4}{(h^2 + L^2/4)^3} + \frac{2 h \dot{y}}{(h^2 + L^2/4)^2} \right)
  \]
\end{frame}

\begin{frame}{Conclusiones}
  \begin{block}{}
    \begin{itemize}
      \item Una pastilla es sensible a las variaciones en el flujo magnético a través de ella
      \item La frecuencia de la oscilación en cada eje determina la frecuencia de la corriente
      \item El mecanismo de la guitarra eléctrica induce armónicos extra que son la causa de su sonido característico.
      \item Podemos utilizar las ecuaciones obtenidas de modelar el mecanismo para recrear los sonidos que generarían otros instrumentos similares.
    \end{itemize}
  \end{block}
\end{frame}
